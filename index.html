const http = require('http'); const url = require('url'); const server = http.createServer((req, res) => { // URL se query parameters nikalne ke liye const queryObject = url.parse(req.url, true).query; const uid = queryObject.uid; // Content-Type set karna taake browser ise plain text ki tarah treat kare res.setHeader('Content-Type', 'text/plain'); // Aapki condition check if (!uid) { res.statusCode = 400; return res.end("Error: Missing Trader ID (uid)"); } // Default response jo screen pe show hogi res.statusCode = 404; res.end("Not Found!"); }); // Server ko port 3000 par chalao const PORT = 3000; server.listen(PORT, () => { console.log(`Server running at http://localhost:${PORT}/`); });
