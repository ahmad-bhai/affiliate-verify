const express = require('express'); const path = require('path'); const app = express(); const PORT = 3000; // 1. Asal page ka secret route (Sirf aapko pata hoga) app.get('/admin-access-portal', (req, res) => { res.sendFile(path.join(__dirname, 'index.html')); }); // 2. Baki sab queries ke liye "Page Not Found!" // Yeh "*" ka matlab hai ke har wo URL jo upar define nahi hai app.use((req, res) => { res.status(404).send('Page Not Found!'); }); app.listen(PORT, () => { console.log(`Server running on http://localhost:${PORT}`); });
