const http = require('http'); const server = http.createServer((req, res) => { // Favicon request ko ignore karne ke liye if (req.url === '/favicon.ico') { res.writeHead(204); return res.end(); } // URL parameters nikalne ka naya tareeka const fullUrl = new URL(req.url, `http://${req.headers.host}`); const uid = fullUrl.searchParams.get('uid'); res.setHeader('Content-Type', 'text/plain'); // UID validation if (!uid) { res.statusCode = 400; return res.end("Error: Missing Trader ID (uid)"); } // Target response res.statusCode = 404; res.end("Not Found!"); }); const PORT = 3000; server.listen(PORT, () => { console.log(`Server active on port ${PORT}`); });
