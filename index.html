const http = require('http'); const server = http.createServer((req, res) => { // Content-Type text/plain rakha hai taake koi HTML rendering na ho res.setHeader('Content-Type', 'text/plain'); // Status 404 set kiya hai aur sirf "Not Found!" send kar raha hoon res.statusCode = 404; res.end("Not Found!"); }); // Port 3000 par server chalega const PORT = 3000; server.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
